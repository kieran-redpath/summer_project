## Get pathway names, and extract mouse-only pathways
rName <- as.list(reactomePATHNAME2ID)
rName <- rName[grep("Mus musculus", names(rName))]

## List of all raectome pathways, and entrez IDs of the genes they contain
rGenes <- as.list(reactomePATHID2EXTID)

## Extract ONLY the mouse pathways 
rGenesPath <- rGenes[match(rName, names(rGenes))]

## Make sure that each gene is only listed once per pathway
rGenesPath <- lapply(rGenesPath, unique)

## Get list of genes, and the pathways they are in.
rGeneByPath <- as.list(reactomeEXTID2PATHID)

## myGenes is the list of all the genes in your data set (e.g., the rownames of 
## your expression matrix): Entrez IDs (here I've just cheated and used all the 
## genes in reactome for mm9)
## allGenes is the intersection of the genes from the data set with the mm9
## reactome genes
myGenes <- unique(unlist(rGenesPath))
allGenes <- intersect( myGenes, unique(unlist(rGenesPath)) )
length(allGenes)
## [1] 8757
## sigGenes is the binary list of whether or ot the gene was significant.
## same length and order as allGenes
## Here I've just randomly generated 0's and 1's
sigGenes <- rep(0, length(allGenes))
names(sigGenes) <- allGenes
